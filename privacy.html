import torch from unsloth import FastLanguageModel from datasets import load_dataset from trl import SFTTrainer from transformers import TrainingArguments print("🚀 3. Ödev: Unsloth ile Model Fine-Tune (LoRA) Eğitimi Başlatılıyor...") # ========================================== # 1. AYARLAR VE PARAMETRELER # ========================================== max_seq_length = 2048 # Modelin tek seferde okuyabileceği maksimum token uzunluğu dtype = None # Otomatik algılama (Float16 veya Bfloat16) load_in_4bit = True # Bellek tasarrufu için modeli 4-bit sıkıştırarak yüklüyoruz # Eğitmek için seçeceğimiz temel model (Unsloth optimize edilmiş Llama modeli) model_name = "unsloth/Llama-3-8B-Instruct" # ========================================== # 2. MODELİ VE TOKENIZER'I YÜKLEME # ========================================== print(f"📥 Temel model yükleniyor: {model_name}...") model, tokenizer = FastLanguageModel.from_pretrained( model_name = model_name, max_seq_length = max_seq_length, dtype = dtype, load_in_4bit = load_in_4bit, ) # ========================================== # 3. LoRA ADAPTÖRLERİNİ EKLEME (PEFT) # ========================================== # Modelin tamamını eğitmek yerine sadece belirli katmanlara (q_proj, v_proj vb.) # LoRA adaptörleri ekleyerek hafif bir eğitim alanı yaratıyoruz. print("⚙️ LoRA adaptörleri modele entegre ediliyor...") model = FastLanguageModel.get_peft_model( model, r = 16, # Rank değeri (Genellikle 8, 16 veya 32 seçilir, ne kadar yüksekse o kadar detay öğrenir) target_modules = ["q_proj", "k_proj", "v_proj", "o_proj", "gate_proj", "up_proj", "down_proj"], lora_alpha = 16, lora_dropout = 0, # Dropout oranı (0 en iyisi ve en kararlısıdır) bias = "none", # Bias optimizasyonu kapalı use_gradient_checkpointing = "unsloth", # Bellek tasarrufu için kritik ayar ) # ========================================== # 4. 1. ÖDEVDE YÜKLEDİĞİMİZ VERİ SETİNİ ÇEKME # ========================================== # Hugging Face profiline kendi yüklediğin veri setinin yolunu buraya yazmalısın: dataset_repo = "Aysenur44/magibu_ai_ozel_veri_seti" print(f"📂 Hugging Face'ten veri seti indiriliyor: {dataset_repo}") dataset = load_dataset(dataset_repo, split="turkish") # ========================================== # 5. EĞİTİCİ (TRAINER) YAPILANDIRMASI # ========================================== # Modelin nasıl eğitileceğini, kaç adım süreceğini ve öğrenme oranını ayarlıyoruz. trainer = SFTTrainer( model = model, tokenizer = tokenizer, train_dataset = dataset, dataset_text_field = "conversations", # Veri setimizdeki anahtar sütun adı max_seq_length = max_seq_length, dataset_num_proc = 2, args = TrainingArguments( per_device_train_batch_size = 2, # Ekran kartı başına düşen paket boyutu gradient_accumulation_steps = 4, # Gradyan biriktirme adımı warmup_steps = 5, # Başlangıç ısınma adımları max_steps = 30, # Ödev için kısa tutulmuş toplam eğitim adımı learning_rate = 2e-4, # Öğrenme hızı (Learning rate) fp16 = not torch.cuda.is_bf16_supported(), bf16 = torch.cuda.is_bf16_supported(), logging_steps = 1, output_dir = "outputs", # Çıktıların geçici saklanacağı yer ), ) # ========================================== # 6. EĞİTİMİ BAŞLATMA # ========================================== print("🔥 Model eğitimi (Fine-Tuning) başladı...") trainer_stats = trainer.train() print("🎉 Model eğitimi başarıyla tamamlandı!") # ========================================== # 7. LoRA ADAPTÖRÜNÜ HUGGING FACE'E YÜKLEME # ========================================== # Çıkan LoRA ağırlıklarını paketleyip Hugging Face profiline gönderiyoruz. lora_repo = "Aysenur44/magibu_ai_lora_adaptor" # Kendi HF repo adın print(f"🌐 LoRA adaptörü Hugging Face'e yükleniyor: {lora_repo}") model.push_to_hub_merged( lora_repo, tokenizer, save_method = "lora", # Sadece LoRA adaptörünü yüklüyoruz token = None # Eğer terminalde login yaptıysan burası None kalabilir ) print("🎯 Tebrikler! 3. Ödev de başarıyla tamamlandı ve LoRA adaptörün yayına alındı!")
